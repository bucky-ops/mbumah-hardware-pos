   125→function useLiveClock() {
   126→  const [now, setNow] = useState(new Date());
   127→
   128→  useEffect(() => {
   129→    const timer = setInterval(() => setNow(new Date()), 60000); // every minute
   130→    return () => clearInterval(timer);
   131→  }, []);
   132→
   133→  return now;
   134→}

   296→interface NotificationItem {
   297→  id: string;
   298→  type: 'low_stock' | 'out_of_stock' | 'overdue_rental' | 'large_debt';
   299→  title: string;
   300→  description: string;
   301→  timestamp: string;
   302→  severity: 'critical' | 'warning' | 'info';
   303→  targetTab?: AppTab;
   304→}
   305→
   306→function NotificationCenter({
   307→  open,
   308→  onOpenChange,
   309→  storeId,
   310→}: {
   311→  open: boolean;
   312→  onOpenChange: (open: boolean) => void;
   313→  storeId: string;
   314→}) {
   315→  const [readIds, setReadIds] = useState<Set<string>>(new Set());
   316→  const { setActiveTab } = useAppStore();
   317→
   318→  const { data: productsData } = useQuery({
   319→    queryKey: ['products-notifications', storeId],
   320→    queryFn: () => productsApi.list({ storeId, limit: 200 }),
   321→    enabled: open,
   322→  });
   323→
   324→  const { data: rentalsData } = useQuery({
   325→    queryKey: ['rentals-notifications', storeId],
   326→    queryFn: () => rentalsApi.list({ storeId, limit: 100 }),
   327→    enabled: open,
   328→  });
   329→
   330→  const { data: debtData } = useQuery({
   331→    queryKey: ['debt-notifications', storeId],
   332→    queryFn: () => debtApi.list({ storeId, limit: 100 }),
   333→    enabled: open,
   334→  });
   335→
   336→  const products = productsData?.data || [];
   337→  const rentals = rentalsData?.data || [];
   338→  const debts = debtData?.data || [];
   339→
   340→  const notifications = useMemo<NotificationItem[]>(() => {
   341→    const items: NotificationItem[] = [];
   342→
   343→    // Out of stock products
   344→    products
   345→      .filter((p) => p.quantityInStock <= 0)
   346→      .forEach((p) => {
   347→        items.push({
   348→          id: `oos-${p.id}`,
   349→          type: 'out_of_stock',
   350→          title: 'Out of Stock',
   351→          description: `${p.name} is completely out of stock`,
   352→          timestamp: p.updatedAt,
   353→          severity: 'critical',
   354→          targetTab: 'inventory',
   355→        });
   356→      });
   357→
   358→    // Low stock products
   359→    products
   360→      .filter((p) => p.quantityInStock > 0 && p.quantityInStock <= p.reorderLevel)
   361→      .forEach((p) => {
   362→        items.push({
   363→          id: `low-${p.id}`,
   364→          type: 'low_stock',
   365→          title: 'Low Stock Alert',
   366→          description: `${p.name} has only ${p.quantityInStock} ${p.unitType.toLowerCase()}s left (reorder at ${p.reorderLevel})`,
   367→          timestamp: p.updatedAt,
   368→          severity: 'warning',
   369→          targetTab: 'inventory',
   370→        });
   371→      });
   372→
   373→    // Overdue rentals
   374→    rentals
   375→      .filter((r) => r.status === 'OVERDUE')
   376→      .forEach((r) => {
   377→        items.push({
   378→          id: `rental-${r.id}`,
   379→          type: 'overdue_rental',
   380→          title: 'Overdue Rental',
   381→          description: `${r.product?.name || 'Equipment'} rented by ${r.customer?.name || 'Customer'} is overdue`,
   382→          timestamp: r.expectedReturnDate,
   383→          severity: 'critical',
   384→          targetTab: 'rentals',
   385→        });
   386→      });
   387→
   388→    // Large outstanding debts (>50,000 KES)
   389→    debts
   390→      .filter((d) => d.balance > 50000 && d.status !== 'SETTLED')
   391→      .forEach((d) => {
   392→        items.push({
   393→          id: `debt-${d.id}`,
   394→          type: 'large_debt',
   395→          title: 'Large Outstanding Debt',
   396→          description: `${d.customer?.name || 'Customer'} owes ${formatKES(d.balance)}`,
   397→          timestamp: d.dueDate,
   398→          severity: 'warning',
   399→          targetTab: 'customers',
   400→        });
   401→      });
   402→
   403→    // Sort by severity (critical first), then by timestamp (newest first)
   404→    items.sort((a, b) => {
   405→      const sevOrder = { critical: 0, warning: 1, info: 2 };
   406→      const sevDiff = sevOrder[a.severity] - sevOrder[b.severity];
   407→      if (sevDiff !== 0) return sevDiff;
   408→      return new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime();
   409→    });
   410→
   411→    return items;
   412→  }, [products, rentals, debts]);
   413→
   414→  const unreadCount = notifications.filter((n) => !readIds.has(n.id)).length;
   415→
   416→  const markAllRead = () => {
   417→    setReadIds(new Set(notifications.map((n) => n.id)));
   418→  };
   419→
   420→  const handleNotificationClick = (notification: NotificationItem) => {
   421→    setReadIds((prev) => new Set([...prev, notification.id]));
   422→    if (notification.targetTab) {
   423→      setActiveTab(notification.targetTab);
   424→      onOpenChange(false);
   425→    }
   426→  };
   427→
   428→  const getSeverityIcon = (type: NotificationItem['type']) => {
   429→    switch (type) {
   430→      case 'out_of_stock': return <PackageX className="h-4 w-4 text-red-500" />;
   431→      case 'low_stock': return <AlertTriangle className="h-4 w-4 text-amber-500" />;
   432→      case 'overdue_rental': return <AlertOctagon className="h-4 w-4 text-red-500" />;
   433→      case 'large_debt': return <CircleDollarSign className="h-4 w-4 text-amber-500" />;
   434→    }
   435→  };
   436→
   437→  const getSeverityBg = (severity: NotificationItem['severity']) => {
   438→    switch (severity) {
   439→      case 'critical': return 'bg-red-50 dark:bg-red-950/20 border-red-200 dark:border-red-900/40';
   440→      case 'warning': return 'bg-amber-50 dark:bg-amber-950/20 border-amber-200 dark:border-amber-900/40';
   441→      case 'info': return 'bg-blue-50 dark:bg-blue-950/20 border-blue-200 dark:border-blue-900/40';
   442→    }
   443→  };
   444→
   445→  return (
   446→    <Sheet open={open} onOpenChange={onOpenChange}>
   447→      <SheetContent side="right" className="w-full sm:max-w-md p-0">
   448→        <SheetHeader className="p-4 pb-2 border-b">
   449→          <div className="flex items-center justify-between">
   450→            <div className="flex items-center gap-2">
   451→              <SheetTitle className="text-lg">Notifications</SheetTitle>
   452→              {unreadCount > 0 && (
   453→                <Badge variant="destructive" className="text-[10px] px-1.5 py-0">
   454→                  {unreadCount}
   455→                </Badge>
   456→              )}
   457→            </div>
   458→            {unreadCount > 0 && (
   459→              <Button variant="ghost" size="sm" className="text-xs h-7" onClick={markAllRead}>
   460→                <CheckCheck className="h-3.5 w-3.5 mr-1" />
   461→                Mark all read
   462→              </Button>
   463→            )}
   464→          </div>
   465→          <SheetDescription>
   466→            Stay updated on stock levels, overdue rentals, and outstanding debts
   467→          </SheetDescription>
   468→        </SheetHeader>
   469→        <ScrollArea className="flex-1 h-[calc(100vh-120px)]">
   470→          {notifications.length === 0 ? (
   471→            <div className="p-8 text-center">
   472→              <BellRing className="h-12 w-12 mx-auto text-muted-foreground/30 mb-3" />
   473→              <p className="text-sm font-medium text-muted-foreground">No notifications</p>
   474→              <p className="text-xs text-muted-foreground/60 mt-1">You&apos;re all caught up!</p>
   475→            </div>
   476→          ) : (
   477→            <div className="p-3 space-y-2">
   478→              {notifications.map((notification) => {
   479→                const isUnread = !readIds.has(notification.id);
   480→                return (
   481→                  <div
   482→                    key={notification.id}
   483→                    role="button"
   484→                    tabIndex={0}
   485→                    onClick={() => handleNotificationClick(notification)}
   486→                    onKeyDown={(e) => { if (e.key === 'Enter' || e.key === ' ') { e.preventDefault(); handleNotificationClick(notification); } }}
   487→                    className={`w-full text-left p-3 rounded-lg border transition-all hover:shadow-sm cursor-pointer ${getSeverityBg(notification.severity)} ${isUnread ? 'ring-1 ring-primary/20' : 'opacity-70'}`}
   488→                  >
   489→                    <div className="flex items-start gap-3">
   490→                      <div className="shrink-0 mt-0.5">{getSeverityIcon(notification.type)}</div>
   491→                      <div className="flex-1 min-w-0">
   492→                        <div className="flex items-center gap-2">

  1034→// Animated counter hook
  1035→function useAnimatedCounter(target: number, duration = 800) {
  1036→  const [count, setCount] = useState(0);
  1037→  const prevTarget = useRef(0);
  1038→
  1039→  useEffect(() => {
  1040→    if (target === prevTarget.current) return;
  1041→    const start = prevTarget.current;
  1042→    const diff = target - start;
  1043→    const startTime = performance.now();
  1044→
  1045→    const animate = (now: number) => {
  1046→      const elapsed = now - startTime;
  1047→      const progress = Math.min(elapsed / duration, 1);
  1048→      // Ease out cubic
  1049→      const eased = 1 - Math.pow(1 - progress, 3);
  1050→      setCount(Math.round(start + diff * eased));
  1051→      if (progress < 1) {
  1052→        requestAnimationFrame(animate);
  1053→      } else {
  1054→        prevTarget.current = target;
  1055→      }
  1056→    };
  1057→
  1058→    requestAnimationFrame(animate);
  1059→  }, [target, duration]);
  1060→
  1061→  return count;
  1062→}

  1065→function MiniSparkline({ data, color, height = 24 }: { data: number[]; color: string; height?: number }) {
  1066→  if (data.length < 2) return null;
  1067→  const max = Math.max(...data);
  1068→  const min = Math.min(...data);
  1069→  const range = max - min || 1;
  1070→  const width = 60;
  1071→  const points = data.map((v, i) => {
  1072→    const x = (i / (data.length - 1)) * width;
  1073→    const y = height - ((v - min) / range) * (height - 4) - 2;
  1074→    return `${x},${y}`;
  1075→  }).join(' ');
  1076→
  1077→  return (
  1078→    <svg width={width} height={height} className="opacity-50 shrink-0" aria-hidden="true">
  1079→      <polyline fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" points={points} />
  1080→    </svg>
  1081→  );

  1247→  const [canScrollRight, setCanScrollRight] = useState(false);
  1248→
  1249→  const checkScroll = useCallback(() => {
  1250→    const el = scrollRef.current;
  1251→    if (!el) return;
  1252→    setCanScrollLeft(el.scrollLeft > 0);

  1257→    checkScroll();
  1258→    const el = scrollRef.current;
  1259→    if (el) {
  1260→      el.addEventListener('scroll', checkScroll);
  1261→      window.addEventListener('resize', checkScroll);
  1262→    }
  1263→    return () => {
  1264→      if (el) el.removeEventListener('scroll', checkScroll);
  1265→      window.removeEventListener('resize', checkScroll);
