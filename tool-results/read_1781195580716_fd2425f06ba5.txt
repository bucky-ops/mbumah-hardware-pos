   344→function POSTab() {
   345→  const [searchQuery, setSearchQuery] = useState('');
   346→  const [selectedCategory, setSelectedCategory] = useState<string>('all');
   347→  const [checkoutOpen, setCheckoutOpen] = useState(false);
   348→  const [mpesaDialogOpen, setMpesaDialogOpen] = useState(false);
   349→  const [mpesaPhone, setMpesaPhone] = useState('');
   350→  const [mpesaStatus, setMpesaStatus] = useState<'idle' | 'processing' | 'success' | 'failed'>('idle');
   351→  const [paymentMethod, setPaymentMethod] = useState<PaymentMethod>('CASH');
   352→  const [cashReceived, setCashReceived] = useState('');
   353→  const [selectedCustomer, setSelectedCustomer] = useState<string>('');
   354→  const currentStoreId = useAppStore((s) => s.currentStoreId);
   355→
   356→  const cart = useCartStore();
   357→  const subtotal = cart.getSubtotal();
   358→  const tax = cart.getTax();
   359→  const total = cart.getTotal();
   360→
   361→  const { data: productsData, isLoading: productsLoading } = useQuery({
   362→    queryKey: ['products', currentStoreId, searchQuery, selectedCategory],
   363→    queryFn: () => productsApi.list({
   364→      storeId: currentStoreId,
   365→      search: searchQuery || undefined,
   366→      limit: 100,
   367→      ...(selectedCategory !== 'all' ? { categoryId: selectedCategory } : {}),
   368→    }),
   369→  });
   370→
   371→  const { data: categoriesData } = useQuery({
   372→    queryKey: ['categories', currentStoreId],
   373→    queryFn: () => categoriesApi.list(currentStoreId),
   374→  });
   375→
   376→  const { data: customersData } = useQuery({
   377→    queryKey: ['customers', currentStoreId],
   378→    queryFn: () => customersApi.list({ storeId: currentStoreId, limit: 100 }),
   379→  });
   380→
   381→  const checkoutMutation = useMutation({
   382→    mutationFn: transactionsApi.create,
   383→    onSuccess: () => {
   384→      toast.success('Transaction completed successfully!');
   385→      cart.clearCart();
   386→      setCheckoutOpen(false);
   387→      setCashReceived('');
   388→      setSelectedCustomer('');
   389→    },
   390→    onError: (err: Error) => {
   391→      toast.error(err.message || 'Checkout failed');
   392→    },
   393→  });
   394→
   395→  const mpesaMutation = useMutation({
   396→    mutationFn: paymentsApi.initiateMpesa,
   397→    onSuccess: (res) => {
   398→      if (res.data) {
   399→        setMpesaStatus('processing');
   400→        const poll = setInterval(async () => {
   401→          try {
   402→            const statusRes = await paymentsApi.getMpesaStatus(res.data!.checkoutRequestId);
   403→            if (statusRes.data?.resultCode === '0') {
   404→              setMpesaStatus('success');
   405→              clearInterval(poll);
   406→            } else if (statusRes.data?.resultCode && statusRes.data.resultCode !== '0') {
   407→              setMpesaStatus('failed');
   408→              clearInterval(poll);
   409→            }
   410→          } catch {
   411→            // keep polling
   412→          }
   413→        }, 3000);
   414→        setTimeout(() => clearInterval(poll), 120000);
   415→      }
   416→    },
   417→    onError: () => {
   418→      setMpesaStatus('failed');
   419→    },
   420→  });
   421→
   422→  const products = productsData?.data || [];
   423→  const categories = categoriesData?.data || [];
   424→  const customers = customersData?.data || [];
   425→
   426→  const handleAddToCart = (product: ProductListItem) => {
   427→    if (product.quantityInStock <= 0 && !product.isRentalItem) {
   428→      toast.error('Product is out of stock');
   429→      return;
   430→    }
   431→    cart.addItem({
   432→      productId: product.id,
   433→      productName: product.name,
   434→      sku: product.sku,
   435→      quantity: 1,
   436→      unitType: product.unitType as UnitType,
   437→      pricePerUnit: product.pricePerUnit,
   438→      costPrice: product.costPrice,
   439→      discountPercent: 0,
   440→      taxRate: product.taxRate,
   441→      isRentalItem: product.isRental,
   442→      isBundle: product.isBundle,
   443→    });
   444→    toast.success(`${product.name} added to cart`);
   445→  };
   446→
   447→  const handleCheckout = () => {
   448→    if (cart.items.length === 0) {
   449→      toast.error('Cart is empty');
   450→      return;
   451→    }
   452→
   453→    if (paymentMethod === 'MPESA') {
   454→      setMpesaDialogOpen(true);
   455→      return;
   456→    }
   457→
   458→    checkoutMutation.mutate({
   459→      storeId: currentStoreId,
   460→      customerId: selectedCustomer || undefined,
   461→      cashierId: useAuthStore.getState().user?.id || '',
   462→      items: cart.items,
   463→      paymentMethod,
   464→      paymentDetails: {
   465→        cashAmount: paymentMethod === 'CASH' ? Number(cashReceived) || total : undefined,
   466→        debtAccountId: paymentMethod === 'DEBT' ? selectedCustomer : undefined,
   467→      },
   468→    });
   469→  };
   470→
   471→  const handleMpesaPay = () => {
   472→    if (!mpesaPhone || mpesaPhone.length < 10) {
   473→      toast.error('Enter a valid phone number');
   474→      return;
   475→    }
   476→    mpesaMutation.mutate({
   477→      phoneNumber: mpesaPhone.startsWith('0') ? `254${mpesaPhone.slice(1)}` : mpesaPhone,
   478→      amount: total,
   479→      accountReference: `MBT-${Date.now()}`,
   480→      transactionDesc: 'MBUMAH HARDWARE Purchase',
   481→    });
   482→  };
   483→
   484→  const change = paymentMethod === 'CASH' && cashReceived ? Number(cashReceived) - total : 0;
   485→
   486→  return (
   487→    <div className="flex flex-col lg:flex-row gap-4 h-full">
   488→      {/* Product Grid */}
   489→      <div className="flex-1 min-w-0 space-y-4">
   490→        {/* Search and Filters */}
   491→        <div className="flex flex-col sm:flex-row gap-3">
   492→          <div className="relative flex-1">
   493→            <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
   494→            <Input
   495→              placeholder="Search products by name, SKU, or barcode..."
   496→              value={searchQuery}
   497→              onChange={(e) => setSearchQuery(e.target.value)}
   498→              className="pl-10"
   499→            />
   500→          </div>
   501→          <Select value={selectedCategory} onValueChange={setSelectedCategory}>
   502→            <SelectTrigger className="w-full sm:w-48">
   503→              <SelectValue placeholder="All Categories" />
   504→            </SelectTrigger>
   505→            <SelectContent>
   506→              <SelectItem value="all">All Categories</SelectItem>
   507→              {categories.map((cat) => (
   508→                <SelectItem key={cat.id} value={cat.id}>{cat.name}</SelectItem>
   509→              ))}
   510→            </SelectContent>
   511→          </Select>
   512→        </div>
   513→
   514→        {/* Products Grid */}
   515→        {productsLoading ? (
   516→          <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-3 xl:grid-cols-4 gap-3">
   517→            {Array.from({ length: 8 }).map((_, i) => (
   518→              <Card key={i} className="overflow-hidden">
   519→                <Skeleton className="h-28" />
   520→                <CardContent className="p-3 space-y-2">
   521→                  <Skeleton className="h-4 w-3/4" />
   522→                  <Skeleton className="h-3 w-1/2" />
   523→                  <Skeleton className="h-5 w-2/3" />
   524→                </CardContent>
   525→              </Card>
   526→            ))}
   527→          </div>
   528→        ) : products.length === 0 ? (
   529→          <div className="text-center py-16">
