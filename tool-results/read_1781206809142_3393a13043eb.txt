   297→   297→// Hook to provide notification count globally
   298→   298→function useNotificationCount(storeId: string) {
   299→   299→  const { data } = useQuery({
   300→   300→    queryKey: ['notification-count', storeId],
   301→   301→    queryFn: async () => {
   302→   302→      const res = await notificationsApi.list(storeId);
   303→   303→      return res.data || [];
   304→   304→    },
   305→   305→    refetchInterval: 60000, // Refresh every minute
   306→   306→    select: (notifications) => {
   307→   307→      const stored = localStorage.getItem('mbt_read_notifications');
   308→   308→      const storedDismissed = localStorage.getItem('mbt_dismissed_notifications');
   309→   309→      const readIds: Set<string> = stored ? new Set(JSON.parse(stored)) : new Set();
   310→   310→      const dismissedIds: Set<string> = storedDismissed ? new Set(JSON.parse(storedDismissed)) : new Set();
   311→   311→      const active = notifications.filter((n) => !dismissedIds.has(n.id));
   312→   312→      const unread = active.filter((n) => !readIds.has(n.id));
   313→   313→      return {
   314→   314→        total: active.length,
   315→   315→        unread: unread.length,
   316→   316→        critical: unread.filter((n) => n.severity === 'critical').length,
   317→   317→        hasNew: unread.length > 0,
   318→   318→      };
   319→   319→    },
   320→   320→  });
   321→   321→  return data || { total: 0, unread: 0, critical: 0, hasNew: false };
   322→   322→}

   326→   326→function NotificationCenter({
   327→   327→  open,
   328→   328→  onOpenChange,
   329→   329→  storeId,
   330→   330→}: {
   331→   331→  open: boolean;
   332→   332→  onOpenChange: (open: boolean) => void;
   333→   333→  storeId: string;
   334→   334→}) {
   335→   335→  const [filter, setFilter] = useState<NotificationFilter>('all');
   336→   336→  const { setActiveTab } = useAppStore();
   337→   337→
   338→   338→  // Persist read/dismissed state in localStorage
   339→   339→  const [readIds, setReadIds] = useState<Set<string>>(() => {
   340→   340→    if (typeof window === 'undefined') return new Set();
   341→   341→    try {
   342→   342→      const stored = localStorage.getItem('mbt_read_notifications');
   343→   343→      return stored ? new Set(JSON.parse(stored)) : new Set();
   344→   344→    } catch {
   345→   345→      return new Set();
   346→   346→    }
   347→   347→  });
   348→   348→
   349→   349→  const [dismissedIds, setDismissedIds] = useState<Set<string>>(() => {
   350→   350→    if (typeof window === 'undefined') return new Set();
   351→   351→    try {
   352→   352→      const stored = localStorage.getItem('mbt_dismissed_notifications');
   353→   353→      return stored ? new Set(JSON.parse(stored)) : new Set();
   354→   354→    } catch {
   355→   355→      return new Set();
   356→   356→    }
   357→   357→  });
   358→   358→
   359→   359→  // Save to localStorage whenever they change
   360→   360→  useEffect(() => {
   361→   361→    try {
   362→   362→      localStorage.setItem('mbt_read_notifications', JSON.stringify([...readIds]));
   363→   363→    } catch { /* ignore */ }
   364→   364→  }, [readIds]);
   365→   365→
   366→   366→  useEffect(() => {
   367→   367→    try {
   368→   368→      localStorage.setItem('mbt_dismissed_notifications', JSON.stringify([...dismissedIds]));
   369→   369→    } catch { /* ignore */ }
   370→   370→  }, [dismissedIds]);
   371→   371→
   372→   372→  const { data: notificationsData, isLoading } = useQuery({
   373→   373→    queryKey: ['notifications', storeId],
   374→   374→    queryFn: async () => {
   375→   375→      const res = await notificationsApi.list(storeId);
   376→   376→      return res.data || [];
   377→   377→    },
   378→   378→    enabled: open,
   379→   379→  });
   380→   380→
   381→   381→  const allNotifications = notificationsData || [];
   382→   382→
   383→   383→  // Filter out dismissed notifications
   384→   384→  const activeNotifications = useMemo(
   385→   385→    () => allNotifications.filter((n) => !dismissedIds.has(n.id)),
   386→   386→    [allNotifications, dismissedIds]
   387→   387→  );
   388→   388→
   389→   389→  // Apply severity filter
   390→   390→  const filteredNotifications = useMemo(() => {
   391→   391→    if (filter === 'all') return activeNotifications;
   392→   392→    return activeNotifications.filter((n) => n.severity === filter);
   393→   393→  }, [activeNotifications, filter]);
   394→   394→
   395→   395→  const unreadCount = activeNotifications.filter((n) => !readIds.has(n.id)).length;
   396→   396→
   397→   397→  // Vibrate on new critical notifications
   398→   398→  useEffect(() => {
   399→   399→    if (open && activeNotifications.length > 0) {
   400→   400→      const criticalUnread = activeNotifications.filter(
   401→   401→        (n) => n.severity === 'critical' && !readIds.has(n.id)
   402→   402→      );
   403→   403→      if (criticalUnread.length > 0 && navigator.vibrate) {
   404→   404→        navigator.vibrate(200);
   405→   405→      }
   406→   406→    }
   407→   407→  }, [open, activeNotifications, readIds]);
   408→   408→
   409→   409→  const markAllRead = () => {
   410→   410→    setReadIds(new Set([...readIds, ...activeNotifications.map((n) => n.id)]));
   411→   411→  };
   412→   412→
   413→   413→  const handleNotificationClick = (notification: NotificationItem) => {
   414→   414→    setReadIds((prev) => new Set([...prev, notification.id]));
   415→   415→    const targetTab = notification.targetTab as AppTab | undefined;
   416→   416→    if (targetTab) {
   417→   417→      setActiveTab(targetTab);

  1097→  1097→// Animated counter hook
  1098→  1098→function useAnimatedCounter(target: number, duration = 800) {
  1099→  1099→  const [count, setCount] = useState(0);
  1100→  1100→  const prevTarget = useRef(0);
  1101→  1101→
  1102→  1102→  useEffect(() => {
  1103→  1103→    if (target === prevTarget.current) return;
  1104→  1104→    const start = prevTarget.current;
  1105→  1105→    const diff = target - start;
  1106→  1106→    const startTime = performance.now();
  1107→  1107→
  1108→  1108→    const animate = (now: number) => {
  1109→  1109→      const elapsed = now - startTime;
  1110→  1110→      const progress = Math.min(elapsed / duration, 1);
  1111→  1111→      // Ease out cubic
  1112→  1112→      const eased = 1 - Math.pow(1 - progress, 3);
  1113→  1113→      setCount(Math.round(start + diff * eased));
  1114→  1114→      if (progress < 1) {
  1115→  1115→        requestAnimationFrame(animate);
  1116→  1116→      } else {
  1117→  1117→        prevTarget.current = target;
  1118→  1118→      }
  1119→  1119→    };
  1120→  1120→
  1121→  1121→    requestAnimationFrame(animate);
  1122→  1122→  }, [target, duration]);
  1123→  1123→
  1124→  1124→  return count;
  1125→  1125→}

  1128→  1128→function MiniSparkline({ data, color, height = 24 }: { data: number[]; color: string; height?: number }) {
  1129→  1129→  if (data.length < 2) return null;
  1130→  1130→  const max = Math.max(...data);
  1131→  1131→  const min = Math.min(...data);
  1132→  1132→  const range = max - min || 1;
  1133→  1133→  const width = 60;
  1134→  1134→  const points = data.map((v, i) => {
  1135→  1135→    const x = (i / (data.length - 1)) * width;
  1136→  1136→    const y = height - ((v - min) / range) * (height - 4) - 2;
  1137→  1137→    return `${x},${y}`;
  1138→  1138→  }).join(' ');
  1139→  1139→
  1140→  1140→  return (
  1141→  1141→    <svg width={width} height={height} className="opacity-50 shrink-0" aria-hidden="true">
  1142→  1142→      <polyline fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" points={points} />
  1143→  1143→    </svg>
  1144→  1144→  );
  1145→  1145→}
