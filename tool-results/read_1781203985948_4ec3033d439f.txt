   122→function useLiveClock() {
   123→  const [now, setNow] = useState(new Date());
   124→
   125→  useEffect(() => {
   126→    const timer = setInterval(() => setNow(new Date()), 60000); // every minute
   127→    return () => clearInterval(timer);
   128→  }, []);
   129→

   292→
   293→interface NotificationItem {
   294→  id: string;
   295→  type: 'low_stock' | 'out_of_stock' | 'overdue_rental' | 'large_debt';
   296→  title: string;
   297→  description: string;
   298→  timestamp: string;
   299→  severity: 'critical' | 'warning' | 'info';
   300→  targetTab?: AppTab;
   301→}
   302→
   303→function NotificationCenter({
   304→  open,
   305→  onOpenChange,
   306→  storeId,
   307→}: {
   308→  open: boolean;
   309→  onOpenChange: (open: boolean) => void;
   310→  storeId: string;
   311→}) {
   312→  const [readIds, setReadIds] = useState<Set<string>>(new Set());
   313→  const { setActiveTab } = useAppStore();
   314→
   315→  const { data: productsData } = useQuery({
   316→    queryKey: ['products-notifications', storeId],
   317→    queryFn: () => productsApi.list({ storeId, limit: 200 }),
   318→    enabled: open,
   319→  });
   320→
   321→  const { data: rentalsData } = useQuery({
   322→    queryKey: ['rentals-notifications', storeId],
   323→    queryFn: () => rentalsApi.list({ storeId, limit: 100 }),
   324→    enabled: open,
   325→  });
   326→
   327→  const { data: debtData } = useQuery({
   328→    queryKey: ['debt-notifications', storeId],
   329→    queryFn: () => debtApi.list({ storeId, limit: 100 }),
   330→    enabled: open,
   331→  });
   332→
   333→  const products = productsData?.data || [];
   334→  const rentals = rentalsData?.data || [];
   335→  const debts = debtData?.data || [];
   336→
   337→  const notifications = useMemo<NotificationItem[]>(() => {
   338→    const items: NotificationItem[] = [];
   339→
   340→    // Out of stock products
   341→    products
   342→      .filter((p) => p.quantityInStock <= 0)
   343→      .forEach((p) => {
   344→        items.push({
   345→          id: `oos-${p.id}`,
   346→          type: 'out_of_stock',
   347→          title: 'Out of Stock',
   348→          description: `${p.name} is completely out of stock`,
   349→          timestamp: p.updatedAt,
   350→          severity: 'critical',
   351→          targetTab: 'inventory',
   352→        });
   353→      });
   354→
   355→    // Low stock products
   356→    products
   357→      .filter((p) => p.quantityInStock > 0 && p.quantityInStock <= p.reorderLevel)
   358→      .forEach((p) => {
   359→        items.push({
   360→          id: `low-${p.id}`,
   361→          type: 'low_stock',
   362→          title: 'Low Stock Alert',
   363→          description: `${p.name} has only ${p.quantityInStock} ${p.unitType.toLowerCase()}s left (reorder at ${p.reorderLevel})`,
   364→          timestamp: p.updatedAt,
   365→          severity: 'warning',
   366→          targetTab: 'inventory',
   367→        });
   368→      });
   369→
   370→    // Overdue rentals
   371→    rentals
   372→      .filter((r) => r.status === 'OVERDUE')
   373→      .forEach((r) => {
   374→        items.push({
   375→          id: `rental-${r.id}`,
   376→          type: 'overdue_rental',
   377→          title: 'Overdue Rental',
   378→          description: `${r.product?.name || 'Equipment'} rented by ${r.customer?.name || 'Customer'} is overdue`,
   379→          timestamp: r.expectedReturnDate,
   380→          severity: 'critical',
   381→          targetTab: 'rentals',
   382→        });
   383→      });
   384→
   385→    // Large outstanding debts (>50,000 KES)
   386→    debts
   387→      .filter((d) => d.balance > 50000 && d.status !== 'SETTLED')
   388→      .forEach((d) => {
   389→        items.push({
   390→          id: `debt-${d.id}`,
   391→          type: 'large_debt',
   392→          title: 'Large Outstanding Debt',
   393→          description: `${d.customer?.name || 'Customer'} owes ${formatKES(d.balance)}`,
   394→          timestamp: d.dueDate,
   395→          severity: 'warning',
   396→          targetTab: 'customers',
   397→        });
   398→      });
   399→
   400→    // Sort by severity (critical first), then by timestamp (newest first)
   401→    items.sort((a, b) => {
   402→      const sevOrder = { critical: 0, warning: 1, info: 2 };
   403→      const sevDiff = sevOrder[a.severity] - sevOrder[b.severity];
   404→      if (sevDiff !== 0) return sevDiff;
   405→      return new Date(b.timestamp).getTime() - new Date(a.timestamp).getTime();
   406→    });
   407→
   408→    return items;
   409→  }, [products, rentals, debts]);
   410→
   411→  const unreadCount = notifications.filter((n) => !readIds.has(n.id)).length;
   412→
   413→  const markAllRead = () => {
   414→    setReadIds(new Set(notifications.map((n) => n.id)));
   415→  };
   416→
   417→  const handleNotificationClick = (notification: NotificationItem) => {
   418→    setReadIds((prev) => new Set([...prev, notification.id]));
   419→    if (notification.targetTab) {
   420→      setActiveTab(notification.targetTab);
   421→      onOpenChange(false);
   422→    }
   423→  };
   424→
   425→  const getSeverityIcon = (type: NotificationItem['type']) => {
   426→    switch (type) {
   427→      case 'out_of_stock': return <PackageX className="h-4 w-4 text-red-500" />;
   428→      case 'low_stock': return <AlertTriangle className="h-4 w-4 text-amber-500" />;
   429→      case 'overdue_rental': return <AlertOctagon className="h-4 w-4 text-red-500" />;
   430→      case 'large_debt': return <CircleDollarSign className="h-4 w-4 text-amber-500" />;
   431→    }
   432→  };
   433→
   434→  const getSeverityBg = (severity: NotificationItem['severity']) => {
   435→    switch (severity) {
   436→      case 'critical': return 'bg-red-50 dark:bg-red-950/20 border-red-200 dark:border-red-900/40';
   437→      case 'warning': return 'bg-amber-50 dark:bg-amber-950/20 border-amber-200 dark:border-amber-900/40';
   438→      case 'info': return 'bg-blue-50 dark:bg-blue-950/20 border-blue-200 dark:border-blue-900/40';
   439→    }
   440→  };
   441→
   442→  return (
   443→    <Sheet open={open} onOpenChange={onOpenChange}>
   444→      <SheetContent side="right" className="w-full sm:max-w-md p-0">
   445→        <SheetHeader className="p-4 pb-2 border-b">
   446→          <div className="flex items-center justify-between">
   447→            <div className="flex items-center gap-2">
   448→              <SheetTitle className="text-lg">Notifications</SheetTitle>
   449→              {unreadCount > 0 && (
   450→                <Badge variant="destructive" className="text-[10px] px-1.5 py-0">
   451→                  {unreadCount}
   452→                </Badge>
   453→              )}
   454→            </div>
   455→            {unreadCount > 0 && (
   456→              <Button variant="ghost" size="sm" className="text-xs h-7" onClick={markAllRead}>
   457→                <CheckCheck className="h-3.5 w-3.5 mr-1" />
   458→                Mark all read
   459→              </Button>
   460→            )}
   461→          </div>
   462→          <SheetDescription>
   463→            Stay updated on stock levels, overdue rentals, and outstanding debts
   464→          </SheetDescription>
   465→        </SheetHeader>
   466→        <ScrollArea className="flex-1 h-[calc(100vh-120px)]">
   467→          {notifications.length === 0 ? (
   468→            <div className="p-8 text-center">
   469→              <BellRing className="h-12 w-12 mx-auto text-muted-foreground/30 mb-3" />
   470→              <p className="text-sm font-medium text-muted-foreground">No notifications</p>
   471→              <p className="text-xs text-muted-foreground/60 mt-1">You&apos;re all caught up!</p>
   472→            </div>
   473→          ) : (
   474→            <div className="p-3 space-y-2">
   475→              {notifications.map((notification) => {
   476→                const isUnread = !readIds.has(notification.id);
   477→                return (
   478→                  <button
   479→                    key={notification.id}
   480→                    type="button"
   481→                    onClick={() => handleNotificationClick(notification)}
   482→                    className={`w-full text-left p-3 rounded-lg border transition-all hover:shadow-sm cursor-pointer ${getSeverityBg(notification.severity)} ${isUnread ? 'ring-1 ring-primary/20' : 'opacity-70'}`}
   483→                  >
   484→                    <div className="flex items-start gap-3">
   485→                      <div className="shrink-0 mt-0.5">{getSeverityIcon(notification.type)}</div>
   486→                      <div className="flex-1 min-w-0">
   487→                        <div className="flex items-center gap-2">
   488→                          <p className="text-sm font-medium">{notification.title}</p>
   489→                          {isUnread && (
   490→                            <span className="w-2 h-2 rounded-full bg-primary shrink-0" />
   491→                          )}
   492→                        </div>
