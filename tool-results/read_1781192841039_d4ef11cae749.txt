   531‚Üí
   532‚Üí```sql
   533‚Üí-- Full-text search index for products
   534‚ÜíCREATE INDEX idx_products_search ON products 
   535‚ÜíUSING GIN (to_tsvector('english', name || ' ' || COALESCE(description, '')));
   536‚Üí
   537‚Üí-- Financial queries optimization
   538‚ÜíCREATE INDEX idx_sales_date_store ON sales_transactions(store_id, created_at DESC);
   539‚ÜíCREATE INDEX idx_sales_payment_method ON sales_transactions(payment_method, created_at);
   540‚ÜíCREATE INDEX idx_journal_entries_date ON journal_entries(organization_id, entry_date DESC);
   541‚Üí
   542‚Üí-- Inventory queries
   543‚ÜíCREATE INDEX idx_warehouse_stock_low ON warehouse_stock 
   544‚ÜíWHERE quantity_on_hand <= (SELECT reorder_point FROM products p WHERE p.id = product_id);
   545‚Üí
   546‚Üí-- Customer queries
   547‚ÜíCREATE INDEX idx_customers_debt ON customers WHERE total_debt > 0;
   548‚Üí
   549‚Üí-- Materialized views for reporting
   550‚ÜíCREATE MATERIALIZED VIEW daily_sales_summary AS
   551‚ÜíSELECT 
   552‚Üí  st.store_id,
   553‚Üí  DATE(st.created_at) as sale_date,
   554‚Üí  st.payment_method,
   555‚Üí  COUNT(*) as transaction_count,
   556‚Üí  SUM(st.total_amount) as total_sales,
   557‚Üí  SUM(st.tax_amount) as total_tax
   558‚ÜíFROM sales_transactions st
   559‚ÜíGROUP BY st.store_id, DATE(st.created_at), st.payment_method;
   560‚Üí
   561‚ÜíCREATE MATERIALIZED VIEW monthly_financial_summary AS
   562‚ÜíSELECT 
   563‚Üí  organization_id,
   564‚Üí  store_id,
   565‚Üí  DATE_TRUNC('month', created_at) as month,
   566‚Üí  SUM(CASE WHEN payment_method = 'cash' THEN total_amount ELSE 0 END) as cash_sales,
   567‚Üí  SUM(CASE WHEN payment_method = 'mpesa' THEN total_amount ELSE 0 END) as mpesa_sales,
   568‚Üí  SUM(CASE WHEN payment_method = 'debt' THEN total_amount ELSE 0 END) as debt_sales,
   569‚Üí  SUM(tax_amount) as total_tax
   570‚ÜíFROM sales_transactions
   571‚ÜíGROUP BY organization_id, store_id, DATE_TRUNC('month', created_at);
   572‚Üí```
