   297→// Hook to provide notification count globally
   298→function useNotificationCount(storeId: string) {
   299→  const { data } = useQuery({
   300→    queryKey: ['notification-count', storeId],
   301→    queryFn: async () => {
   302→      const res = await notificationsApi.list(storeId);
   303→      return res.data || [];
   304→    },
   305→    refetchInterval: 60000, // Refresh every minute
   306→    select: (notifications) => {
   307→      const stored = localStorage.getItem('mbt_read_notifications');
   308→      const storedDismissed = localStorage.getItem('mbt_dismissed_notifications');
   309→      const readIds: Set<string> = stored ? new Set(JSON.parse(stored)) : new Set();
   310→      const dismissedIds: Set<string> = storedDismissed ? new Set(JSON.parse(storedDismissed)) : new Set();
   311→      const active = notifications.filter((n) => !dismissedIds.has(n.id));
   312→      const unread = active.filter((n) => !readIds.has(n.id));
   313→      return {
   314→        total: active.length,
   315→        unread: unread.length,
   316→        critical: unread.filter((n) => n.severity === 'critical').length,
   317→        hasNew: unread.length > 0,
   318→      };
   319→    },
   320→  });
   321→  return data || { total: 0, unread: 0, critical: 0, hasNew: false };
   322→}

   326→function NotificationCenter({
   327→  open,
   328→  onOpenChange,
   329→  storeId,
   330→}: {
   331→  open: boolean;
   332→  onOpenChange: (open: boolean) => void;
   333→  storeId: string;
   334→}) {
   335→  const [filter, setFilter] = useState<NotificationFilter>('all');
   336→  const { setActiveTab } = useAppStore();
   337→
   338→  // Persist read/dismissed state in localStorage
   339→  const [readIds, setReadIds] = useState<Set<string>>(() => {
   340→    if (typeof window === 'undefined') return new Set();
   341→    try {
   342→      const stored = localStorage.getItem('mbt_read_notifications');
   343→      return stored ? new Set(JSON.parse(stored)) : new Set();
   344→    } catch {
   345→      return new Set();
   346→    }
   347→  });
   348→
   349→  const [dismissedIds, setDismissedIds] = useState<Set<string>>(() => {
   350→    if (typeof window === 'undefined') return new Set();
   351→    try {
   352→      const stored = localStorage.getItem('mbt_dismissed_notifications');
   353→      return stored ? new Set(JSON.parse(stored)) : new Set();
   354→    } catch {
   355→      return new Set();
   356→    }
   357→  });
   358→
   359→  // Save to localStorage whenever they change
   360→  useEffect(() => {
   361→    try {
   362→      localStorage.setItem('mbt_read_notifications', JSON.stringify([...readIds]));
   363→    } catch { /* ignore */ }
   364→  }, [readIds]);
   365→
   366→  useEffect(() => {
   367→    try {
   368→      localStorage.setItem('mbt_dismissed_notifications', JSON.stringify([...dismissedIds]));
   369→    } catch { /* ignore */ }
   370→  }, [dismissedIds]);
   371→
   372→  const { data: notificationsData, isLoading } = useQuery({
   373→    queryKey: ['notifications', storeId],
   374→    queryFn: async () => {
   375→      const res = await notificationsApi.list(storeId);
   376→      return res.data || [];
   377→    },
   378→    enabled: open,
   379→  });
   380→
   381→  const allNotifications = notificationsData || [];
   382→
   383→  // Filter out dismissed notifications
   384→  const activeNotifications = useMemo(
   385→    () => allNotifications.filter((n) => !dismissedIds.has(n.id)),
   386→    [allNotifications, dismissedIds]
   387→  );
   388→
   389→  // Apply severity filter
   390→  const filteredNotifications = useMemo(() => {
   391→    if (filter === 'all') return activeNotifications;
   392→    return activeNotifications.filter((n) => n.severity === filter);
   393→  }, [activeNotifications, filter]);
   394→
   395→  const unreadCount = activeNotifications.filter((n) => !readIds.has(n.id)).length;
   396→
   397→  // Vibrate on new critical notifications
   398→  useEffect(() => {
   399→    if (open && activeNotifications.length > 0) {
   400→      const criticalUnread = activeNotifications.filter(
   401→        (n) => n.severity === 'critical' && !readIds.has(n.id)
   402→      );
   403→      if (criticalUnread.length > 0 && navigator.vibrate) {
   404→        navigator.vibrate(200);
   405→      }
   406→    }
   407→  }, [open, activeNotifications, readIds]);
   408→
   409→  const markAllRead = () => {
   410→    setReadIds(new Set([...readIds, ...activeNotifications.map((n) => n.id)]));
   411→  };
   412→
   413→  const handleNotificationClick = (notification: NotificationItem) => {
   414→    setReadIds((prev) => new Set([...prev, notification.id]));
   415→    const targetTab = notification.targetTab as AppTab | undefined;
   416→    if (targetTab) {
   417→      setActiveTab(targetTab);
   418→      onOpenChange(false);
   419→    }
   420→  };
   421→
   422→  const handleDismiss = (e: React.MouseEvent, notificationId: string) => {
   423→    e.stopPropagation();
   424→    setDismissedIds((prev) => new Set([...prev, notificationId]));
   425→  };
   426→
   427→  const getNotificationIcon = (type: NotificationItem['type']) => {
   428→    switch (type) {
   429→      case 'out_of_stock': return <PackageX className="h-4 w-4 text-red-500" />;
   430→      case 'low_stock': return <AlertTriangle className="h-4 w-4 text-amber-500" />;
   431→      case 'overdue_rental': return <AlertOctagon className="h-4 w-4 text-red-500" />;
   432→      case 'large_debt': return <CircleDollarSign className="h-4 w-4 text-amber-500" />;
   433→      case 'new_customer': return <UserPlus className="h-4 w-4 text-green-500" />;
   434→      case 'recent_transaction': return <Receipt className="h-4 w-4 text-blue-500" />;
   435→    }
   436→  };
   437→
   438→  const getSeverityBg = (severity: NotificationItem['severity']) => {
   439→    switch (severity) {
   440→      case 'critical': return 'bg-red-50 dark:bg-red-950/20 border-red-200 dark:border-red-900/40';
   441→      case 'warning': return 'bg-amber-50 dark:bg-amber-950/20 border-amber-200 dark:border-amber-900/40';
   442→      case 'info': return 'bg-green-50/50 dark:bg-green-950/10 border-green-200 dark:border-green-900/30';
   443→    }
   444→  };
   445→
   446→  const filterTabs: { id: NotificationFilter; label: string; count: number }[] = [
   447→    { id: 'all', label: 'All', count: activeNotifications.length },
   448→    { id: 'critical', label: 'Critical', count: activeNotifications.filter((n) => n.severity === 'critical').length },
   449→    { id: 'warning', label: 'Warnings', count: activeNotifications.filter((n) => n.severity === 'warning').length },
   450→    { id: 'info', label: 'Info', count: activeNotifications.filter((n) => n.severity === 'info').length },
   451→  ];
   452→
   453→  return (
   454→    <Sheet open={open} onOpenChange={onOpenChange}>
   455→      <SheetContent side="right" className="w-full sm:max-w-md p-0 flex flex-col">
   456→        <SheetHeader className="p-4 pb-2 border-b shrink-0">
   457→          <div className="flex items-center justify-between">
   458→            <div className="flex items-center gap-2">
   459→              <SheetTitle className="text-lg">Notifications</SheetTitle>
   460→              {unreadCount > 0 && (
   461→                <Badge variant="destructive" className="text-[10px] px-1.5 py-0 animate-pulse-slow">
   462→                  {unreadCount}
   463→                </Badge>
   464→              )}
   465→            </div>
   466→            <div className="flex items-center gap-1">
   467→              {dismissedIds.size > 0 && (
   468→                <Button
   469→                  variant="ghost"
   470→                  size="sm"
   471→                  className="text-xs h-7 text-muted-foreground"
   472→                  onClick={() => setDismissedIds(new Set())}
   473→                >
   474→                  Show dismissed
   475→                </Button>
   476→              )}
   477→              {unreadCount > 0 && (
   478→                <Button variant="ghost" size="sm" className="text-xs h-7" onClick={markAllRead}>
   479→                  <CheckCheck className="h-3.5 w-3.5 mr-1" />
   480→                  Mark all read
   481→                </Button>
   482→              )}
   483→            </div>
   484→          </div>
   485→          <SheetDescription>
   486→            Stay updated on stock levels, rentals, debts, and more
   487→          </SheetDescription>
   488→        </SheetHeader>

  1097→// Animated counter hook
  1098→function useAnimatedCounter(target: number, duration = 800) {
  1099→  const [count, setCount] = useState(0);
  1100→  const prevTarget = useRef(0);
  1101→
  1102→  useEffect(() => {
  1103→    if (target === prevTarget.current) return;
  1104→    const start = prevTarget.current;
  1105→    const diff = target - start;
  1106→    const startTime = performance.now();
  1107→
  1108→    const animate = (now: number) => {
  1109→      const elapsed = now - startTime;
  1110→      const progress = Math.min(elapsed / duration, 1);
  1111→      // Ease out cubic
  1112→      const eased = 1 - Math.pow(1 - progress, 3);
  1113→      setCount(Math.round(start + diff * eased));
  1114→      if (progress < 1) {
  1115→        requestAnimationFrame(animate);
  1116→      } else {
  1117→        prevTarget.current = target;
  1118→      }
  1119→    };
  1120→
  1121→    requestAnimationFrame(animate);
  1122→  }, [target, duration]);
  1123→
  1124→  return count;
  1125→}

  1128→function MiniSparkline({ data, color, height = 24 }: { data: number[]; color: string; height?: number }) {
  1129→  if (data.length < 2) return null;
  1130→  const max = Math.max(...data);
  1131→  const min = Math.min(...data);
  1132→  const range = max - min || 1;
  1133→  const width = 60;
  1134→  const points = data.map((v, i) => {
  1135→    const x = (i / (data.length - 1)) * width;
  1136→    const y = height - ((v - min) / range) * (height - 4) - 2;
  1137→    return `${x},${y}`;
  1138→  }).join(' ');
  1139→
  1140→  return (
  1141→    <svg width={width} height={height} className="opacity-50 shrink-0" aria-hidden="true">
  1142→      <polyline fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" points={points} />
  1143→    </svg>
  1144→  );

  1299→function CategoryChips({
  1300→  categories,

  1310→  const [canScrollRight, setCanScrollRight] = useState(false);
  1311→
  1312→  const checkScroll = useCallback(() => {
  1313→    const el = scrollRef.current;
  1314→    if (!el) return;
  1315→    setCanScrollLeft(el.scrollLeft > 0);

  1318→
  1319→  useEffect(() => {
  1320→    checkScroll();
  1321→    const el = scrollRef.current;
  1322→    if (el) {
  1323→      el.addEventListener('scroll', checkScroll);
  1324→      window.addEventListener('resize', checkScroll);
  1325→    }
  1326→    return () => {
  1327→      if (el) el.removeEventListener('scroll', checkScroll);
  1328→      window.removeEventListener('resize', checkScroll);
  1329→    };
