   978‚Üí   978‚Üíexport function formatRelativeTime(date: string | Date): string {
   979‚Üí   979‚Üí  const now = new Date();
   980‚Üí   980‚Üí  const then = new Date(date);
   981‚Üí   981‚Üí  const diffMs = now.getTime() - then.getTime();
   982‚Üí   982‚Üí  const diffSec = Math.floor(diffMs / 1000);
   983‚Üí   983‚Üí  const diffMin = Math.floor(diffSec / 60);
   984‚Üí   984‚Üí  const diffHr = Math.floor(diffMin / 60);
   985‚Üí   985‚Üí  const diffDay = Math.floor(diffHr / 24);
   986‚Üí   986‚Üí  const diffWeek = Math.floor(diffDay / 7);
   987‚Üí   987‚Üí
   988‚Üí   988‚Üí  if (diffSec < 60) return 'just now';
   989‚Üí   989‚Üí  if (diffMin < 60) return `${diffMin} min${diffMin !== 1 ? 's' : ''} ago`;
   990‚Üí   990‚Üí  if (diffHr < 24) return `${diffHr} hour${diffHr !== 1 ? 's' : ''} ago`;
   991‚Üí   991‚Üí  if (diffDay < 7) return `${diffDay} day${diffDay !== 1 ? 's' : ''} ago`;
   992‚Üí   992‚Üí  if (diffWeek < 4) return `${diffWeek} week${diffWeek !== 1 ? 's' : ''} ago`;
   993‚Üí   993‚Üí  return formatDate(date);
   994‚Üí   994‚Üí}
